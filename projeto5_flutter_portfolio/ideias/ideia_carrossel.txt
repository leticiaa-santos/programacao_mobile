üéØ Exemplo de Tela com Carrossel (com estado)
import 'package:flutter/material.dart';
import 'package:carousel_slider/carousel_slider.dart';

class Ferramenta {
  final String nome;
  final String descricao;
  final String iconePath;

  Ferramenta({required this.nome, required this.descricao, required this.iconePath});
}

class TelaFerramentas extends StatefulWidget {
  const TelaFerramentas({Key? key}) : super(key: key);

  @override
  State<TelaFerramentas> createState() => _TelaFerramentasState();
}

class _TelaFerramentasState extends State<TelaFerramentas> {
  int _indiceAtual = 0;

  final List<Ferramenta> ferramentas = [
    Ferramenta(nome: "Flutter", descricao: "Framework para apps multiplataforma", iconePath: "assets/flutter.png"),
    Ferramenta(nome: "Firebase", descricao: "Backend como servi√ßo", iconePath: "assets/firebase.png"),
    Ferramenta(nome: "Git", descricao: "Controle de vers√£o", iconePath: "assets/git.png"),
    Ferramenta(nome: "Figma", descricao: "Design de interfaces", iconePath: "assets/figma.png"),
    // Adicione mais se quiser
  ];

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text("Minhas Ferramentas"),
      ),
      body: Column(
        children: [
          const SizedBox(height: 20),
          CarouselSlider.builder(
            itemCount: ferramentas.length,
            itemBuilder: (context, index, realIdx) {
              final ferramenta = ferramentas[index];
              final isActive = index == _indiceAtual;

              return AnimatedContainer(
                duration: const Duration(milliseconds: 300),
                margin: EdgeInsets.symmetric(horizontal: isActive ? 8 : 16, vertical: isActive ? 10 : 20),
                decoration: BoxDecoration(
                  color: isActive ? Colors.blue.shade100 : Colors.grey.shade200,
                  borderRadius: BorderRadius.circular(16),
                  boxShadow: isActive
                      ? [BoxShadow(color: Colors.blue.shade300, blurRadius: 10, offset: Offset(0, 4))]
                      : [],
                ),
                child: Column(
                  mainAxisAlignment: MainAxisAlignment.center,
                  children: [
                    Image.asset(ferramenta.iconePath, width: 80, height: 80),
                    const SizedBox(height: 12),
                    Text(ferramenta.nome, style: const TextStyle(fontSize: 20, fontWeight: FontWeight.bold)),
                    const SizedBox(height: 6),
                    Padding(
                      padding: const EdgeInsets.symmetric(horizontal: 16),
                      child: Text(
                        ferramenta.descricao,
                        style: const TextStyle(fontSize: 14, color: Colors.black54),
                        textAlign: TextAlign.center,
                      ),
                    ),
                  ],
                ),
              );
            },
            options: CarouselOptions(
              height: 300,
              enlargeCenterPage: true,
              onPageChanged: (index, reason) {
                setState(() {
                  _indiceAtual = index;
                });
              },
            ),
          ),
          const SizedBox(height: 16),
          // Indicadores
          Row(
            mainAxisAlignment: MainAxisAlignment.center,
            children: ferramentas.asMap().entries.map((entry) {
              return Container(
                width: 10,
                height: 10,
                margin: const EdgeInsets.symmetric(horizontal: 4),
                decoration: BoxDecoration(
                  shape: BoxShape.circle,
                  color: _indiceAtual == entry.key ? Colors.blueAccent : Colors.grey,
                ),
              );
            }).toList(),
          ),
        ],
      ),
    );
  }
}
